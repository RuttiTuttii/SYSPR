Разработка схем с использованием таймеров
5.1 Каждую секунду выводить в Serial количество секунд
Компоненты: Arduino Uno (ничего больше не нужно)
Подключение: —
Код:
unsigned long previousMillis = 0;
const long interval = 1000;
int seconds = 0;

void setup() {
  Serial.begin(9600);
  Serial.println("Схема запущена");
}

void loop() {
  unsigned long currentMillis = millis();
  if (currentMillis - previousMillis >= interval) {
    previousMillis = currentMillis;
    seconds++;
    Serial.print("Прошло секунд: ");
    Serial.println(seconds);
  }
}

5.2 Синий LED мигает каждые 2 сек, красный — каждые 1,5 сек
Компоненты: • 1× синий LED • 1× красный LED • 2× резистор 220–330 Ом
Подключение: Синий → пин 9 (через резистор) Красный → пин 10 (через резистор) Катоды обоих — на GND
PtVLo“LARGE”
Код:
unsigned long prevBlue = 0, prevRed = 0;
const long intBlue = 2000, intRed = 1500;
bool stBlue = false, stRed = false;

void setup() {
  pinMode(9, OUTPUT);
  pinMode(10, OUTPUT);
}

void loop() {
  unsigned long cur = millis();
  if (cur - prevBlue >= intBlue) { prevBlue = cur; stBlue = !stBlue; digitalWrite(9, stBlue); }
  if (cur - prevRed >= intRed)   { prevRed = cur;  stRed  = !stRed;  digitalWrite(10, stRed); }
}

5.3 LED горит первые 5 сек, потом гаснет + каждую секунду «Hello World»
Компоненты: 1× LED + резистор 220 Ом (или встроенный на 13)
Подключение: LED на пин 8
Код:
unsigned long lastPrint = 0;

void setup() {
  pinMode(8, OUTPUT);
  digitalWrite(8, HIGH);
  Serial.begin(9600);
}

void loop() {
  if (millis() < 5000) digitalWrite(8, HIGH);
  else                 digitalWrite(8, LOW);

  if (millis() - lastPrint >= 1000) {
    lastPrint += 1000;
    Serial.println("Hello World");
  }
}

5.4 PIR-датчик движения → освещение 30 сек (с продлением)
Компоненты: • Модуль HC-SR501 PIR • 1× LED + резистор 220 Ом
Подключение: PIR: VCC → 5V, GND → GND, OUT → пин 2 LED → пин 9
k6P9B“LARGE”
Код:
const int pir = 2, led = 9;
unsigned long offTime = 0;

void setup() {
  pinMode(pir, INPUT);
  pinMode(led, OUTPUT);
}

void loop() {
  if (digitalRead(pir) == HIGH) {
    digitalWrite(led, HIGH);
    offTime = millis() + 30000;
  }
  if (millis() > offTime) digitalWrite(led, LOW);
}

5.5 Меняющаяся частота сигнала каждые 5 сек
Компоненты: 1× LED + резистор
Подключение: LED → пин 11
Код:
const int led = 11;
unsigned long lastToggle = 0, phaseStart = 0;
int interval = 1000;

void setup() { pinMode(led, OUTPUT); phaseStart = millis(); }

void loop() {
  unsigned long now = millis();
  if (now - phaseStart >= 5000) {
    phaseStart = now;
    int p = (now / 5000) % 3;
    interval = (p==0)?1000:(p==1)?750:500;
  }
  if (now - lastToggle >= interval) {
    lastToggle = now;
    digitalWrite(led, !digitalRead(led));
  }
}

5.2 При нажатии кнопки — время + счётчик нажатий в Serial
Код:
const int btn = 2;
unsigned long count = 0, last = 0;

void setup() {
  pinMode(btn, INPUT_PULLUP);
  Serial.begin(9600);
}

void loop() {
  if (digitalRead(btn) == LOW && millis() - last > 50) {
    last = millis();
    count++;
    Serial.print("Время: "); Serial.print(millis()/1000);
    Serial.print(" сек | Нажатий: "); Serial.println(count);
  }
}

5.3 Две кнопки — два LED (переключение)
Подключение: Кнопка1 → 2, Кнопка2 → 3 LED1 → 8, LED2 → 9
KiT7O“LARGE”
Код:
const int b1=2, b2=3, l1=8, l2=9;
void setup() {
  pinMode(b1,INPUT_PULLUP); pinMode(b2,INPUT_PULLUP);
  pinMode(l1,OUTPUT); pinMode(l2,OUTPUT);
  digitalWrite(l1,HIGH);
}
void loop() {
  if (!digitalRead(b1)) { digitalWrite(l1,HIGH); digitalWrite(l2,LOW); delay(200); }
  if (!digitalRead(b2)) { digitalWrite(l1,LOW);  digitalWrite(l2,HIGH); delay(200); }
}

5.4 Прерывание FALLING — пешеходный светофор
Подключение: Красный → 8, Зелёный → 9 Кнопка пешехода → 2 (прерывание 0)
EWTm4“LARGE”
Код:
const int red=8, green=9, btn=2;
volatile bool req = false;

void isr() { req = true; }

void setup() {
  pinMode(red,OUTPUT); pinMode(green,OUTPUT); pinMode(btn,INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(btn), isr, FALLING);
  digitalWrite(red,HIGH);
}

void loop() {
  if (req) {
    req = false;
    digitalWrite(red,LOW); delay(3000);
    for(int i=0;i<3;i++) { digitalWrite(green,HIGH); delay(300); digitalWrite(green,LOW); delay(300); }
    digitalWrite(green,HIGH); delay(10000);
    digitalWrite(green,LOW); digitalWrite(red,HIGH);
  }
}

5.5 Прерывание CHANGE — переключение режима мигания долгим нажатием (≥3 сек)
Подключение: Кнопка → 2, LED → 8
lvVBg“LARGE”
Код:
const int led=8, btn=2;
volatile unsigned long pressStart = 0;
volatile bool longPress = false;
int mode = 0; // 0=быстро, 1=медленно

void isr() {
  if (digitalRead(btn) == LOW) pressStart = millis();
  else if (millis() - pressStart >= 3000) longPress = true;
}

void setup() {
  pinMode(led,OUTPUT); pinMode(btn,INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(btn), isr, CHANGE);
}

void loop() {
  if (longPress) { longPress = false; mode = !mode; }
  int t = (mode==0) ? 200 : 800;
  digitalWrite(led,HIGH); delay(t);
  digitalWrite(led,LOW);  delay(t);
}
